    ```typescript
    import { NextRequest, NextResponse } from 'next/server';
    import { z } from 'zod';
    import * as Sentry from '@sentry/nextjs';
    import { prisma } from '@mpstats/db/client';
    import { createClient } from '@/lib/supabase/server';
    import { notifyCommentReply } from '@/lib/notifications/notify';

    export const dynamic = 'force-dynamic';

    const InputSchema = z.object({
      replyCommentId: z.string().min(1),
    });

    export async function POST(request: NextRequest) {
      try {
        // 1) Supabase auth — реальный pattern из проекта (apps/web/src/app/(main)/layout.tsx)
        const supabase = await createClient();
        const { data } = await supabase.auth.getUser();
        const user = data?.user;
        if (!user) {
          return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
        }

        // 2) Validate input
        const body = await request.json();
        const parsed = InputSchema.safeParse(body);
        if (!parsed.success) {
          return NextResponse.json(
            { error: 'Invalid input', details: parsed.error.flatten() },
            { status: 400 },
          );
        }
        const { replyCommentId } = parsed.data;

        // 3) Anti-spoofing: replier MUST be ctx.user
        const reply = await prisma.lessonComment.findUnique({
          where: { id: replyCommentId },
          select: { id: true, userId: true, parentId: true },
        });
        if (!reply) {
          return NextResponse.json({ error: 'Reply comment not found' }, { status: 404 });
        }
        if (reply.userId !== user.id) {
          // Defensive: someone is trying to trigger notification claiming to be the replier
          Sentry.captureMessage('notify-reply spoofing attempt', {
            level: 'warning',
            extra: { actualReplier: reply.userId, requestingUser: user.id, replyCommentId },
          });
          return NextResponse.json({ error: 'Forbidden' }, { status: 403 });
        }
        if (!reply.parentId) {
          // Not a reply (root comment) — silently no-op
          return NextResponse.json({ ok: true, skipped: 'not_a_reply' });
        }

        // 4) Trigger notification (fire-and-forget; notify() handles its own errors)
        await notifyCommentReply({ replyCommentId, actorUserId: user.id });

        return NextResponse.json({ ok: true });
      } catch (error) {
        Sentry.captureException(error, {
          tags: { area: 'notifications', stage: 'notify-reply-route' },
        });
        console.error('[notify-reply] error:', error);
        // Возвращаем 500 — frontend всё равно не повторит (fire-and-forget contract)
        return NextResponse.json({ ok: false }, { status: 500 });
      }
    }
    ```

    КРИТИЧЕСКИ ВАЖНО:
    - `createClient` импортируется из `@/lib/supabase/server` (verified file content). Это **async function**, не named class — `await createClient()`.
    - `supabase.auth.getUser()` возвращает `{ data: { user: User | null }, error }` — деструктурируй `data?.user`.
    - НЕ изобретать helper'ы вроде `getAuthenticatedUser` — переиспользовать существующий `createClient`.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → /api/notifications/notify-reply | Untrusted: любой залогиненный юзер может вызвать с любым replyCommentId |
| Route handler → notifyCommentReply | Trusted: actorUserId передаётся из ctx.user.id, не из тела запроса |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-51-04-01 | S (Spoofing) | Юзер X вызывает notify-reply с replyCommentId юзера Y | mitigate | findUnique replyCommentId → if reply.userId !== ctx.user.id → 403; explicit unit test |
| T-51-04-02 | S (Spoofing) | Юзер передаёт actorUserId в теле запроса | mitigate | Body schema принимает ТОЛЬКО replyCommentId; actorUserId всегда = ctx.user.id (server-resolved) |
| T-51-04-03 | I (Info disclosure) | 404/403 leak о существовании комментов | accept | Симметрия с tRPC comments router (404 для missing, 403 для existing-not-yours) — стандартная практика |
| T-51-04-04 | D (DoS) | Юзер бомбит /api/notifications/notify-reply | accept | Frontend вызывает только onSuccess после tRPC create; tRPC create уже rate-limited через Vercel/Next.js (если есть); dup notifications защищены отсутствием дедупа на текущем этапе — accept |
| T-51-04-05 | T (Tampering) | replyCommentId не существует | mitigate | findUnique returns null → 404 без notify-call |
| T-51-04-06 | I (Info disclosure) | Replier knows about parent comment author | accept | Это нормально: ответил на коммент Y → автор Y получит уведомление; никаких дополнительных данных не утечёт |
