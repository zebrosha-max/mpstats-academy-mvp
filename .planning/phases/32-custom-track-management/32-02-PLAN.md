**Step 2: Add tRPC utils and mutations** inside the `LearnPage` component, after existing queries:

```typescript
const utils = trpc.useUtils();

const addToTrackMutation = trpc.learning.addToTrack.useMutation({
  onMutate: async ({ lessonId }) => {
    await utils.learning.getRecommendedPath.cancel();
    const prev = utils.learning.getRecommendedPath.getData();
    // Optimistic: add to custom section
    utils.learning.getRecommendedPath.setData(undefined, (old) => {
      if (!old) return old;
      // Clone sections
      const sections = old.sections ? [...old.sections.map(s => ({ ...s, lessons: [...s.lessons], lessonIds: s.lessonIds ? [...s.lessonIds] : undefined }))] : [];
      // Find or create custom section
      let customIdx = sections.findIndex(s => s.id === 'custom');
      if (customIdx < 0) {
        sections.unshift({ id: 'custom', title: 'Мои уроки', description: '0', lessons: [], lessonIds: [] } as any);
        customIdx = 0;
      }
      // Remove from AI sections
      for (let i = 0; i < sections.length; i++) {
        if (sections[i].id !== 'custom') {
          sections[i] = { ...sections[i], lessons: sections[i].lessons.filter((l: any) => l.id !== lessonId) };
        }
      }
      // Add lesson placeholder to custom (will be hydrated on invalidate)
      const alreadyExists = sections[customIdx].lessons.some((l: any) => l.id === lessonId);
      if (!alreadyExists) {
        // Try to find lesson data from courses
        const lessonData = courses?.flatMap(c => c.lessons).find(l => l.id === lessonId);
        if (lessonData) {
          sections[customIdx] = { ...sections[customIdx], lessons: [...sections[customIdx].lessons, lessonData as any] };
        }
      }
      return { ...old, sections, isSectioned: true as const };
    });
    return { prev };
  },
  onError: (_err, _vars, ctx) => {
    if (ctx?.prev) utils.learning.getRecommendedPath.setData(undefined, ctx.prev);
    toast.error('Не удалось добавить урок');
  },
  onSuccess: () => toast.success('Добавлено в трек'),
  onSettled: () => utils.learning.getRecommendedPath.invalidate(),
});

const removeFromTrackMutation = trpc.learning.removeFromTrack.useMutation({
  onMutate: async ({ lessonId }) => {
    await utils.learning.getRecommendedPath.cancel();
    const prev = utils.learning.getRecommendedPath.getData();
    utils.learning.getRecommendedPath.setData(undefined, (old) => {
      if (!old || !old.sections) return old;
      const sections = old.sections.map(s => ({
        ...s,
        lessons: s.lessons.filter((l: any) => l.id !== lessonId),
      }));
      return { ...old, sections };
    });
    return { prev };
  },
  onError: (_err, _vars, ctx) => {
    if (ctx?.prev) utils.learning.getRecommendedPath.setData(undefined, ctx.prev);
    toast.error('Не удалось убрать урок');
  },
  onSuccess: () => toast.success('Убрано из трека'),
  onSettled: () => utils.learning.getRecommendedPath.invalidate(),
});

const rebuildTrackMutation = trpc.learning.rebuildTrack.useMutation({
  onSuccess: () => {
    toast.success('Трек перестроен');
    utils.learning.getRecommendedPath.invalidate();
  },
  onError: () => toast.error('Не удалось перестроить трек'),
});
```
