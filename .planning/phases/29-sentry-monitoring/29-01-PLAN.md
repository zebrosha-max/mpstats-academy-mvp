1. Install @sentry/nextjs:
   ```bash
   cd apps/web && pnpm add @sentry/nextjs
   ```

2. Create `apps/web/sentry.client.config.ts`:
   ```typescript
   import * as Sentry from '@sentry/nextjs';

   Sentry.init({
     dsn: process.env.NEXT_PUBLIC_SENTRY_DSN,
     tracesSampleRate: 0.3,
     replaysSessionSampleRate: 0,
     replaysOnErrorSampleRate: 1.0,
     integrations: [
       Sentry.replayIntegration(),
     ],
     // Don't send events in development
     enabled: process.env.NODE_ENV === 'production',
   });
   ```

3. Create `apps/web/sentry.server.config.ts`:
   ```typescript
   import * as Sentry from '@sentry/nextjs';

   Sentry.init({
     dsn: process.env.NEXT_PUBLIC_SENTRY_DSN,
     tracesSampleRate: 0.3,
     enabled: process.env.NODE_ENV === 'production',
   });
   ```

4. Create `apps/web/sentry.edge.config.ts`:
   ```typescript
   import * as Sentry from '@sentry/nextjs';

   Sentry.init({
     dsn: process.env.NEXT_PUBLIC_SENTRY_DSN,
     tracesSampleRate: 0.3,
     enabled: process.env.NODE_ENV === 'production',
   });
   ```

5. Create `apps/web/src/instrumentation.ts`:
   ```typescript
   import * as Sentry from '@sentry/nextjs';

   export async function register() {
     if (process.env.NEXT_RUNTIME === 'nodejs') {
       await import('../sentry.server.config');
     }
     if (process.env.NEXT_RUNTIME === 'edge') {
       await import('../sentry.edge.config');
     }
   }

   export const onRequestError = Sentry.captureRequestError;
   ```

6. Update `apps/web/src/app/global-error.tsx`:
   - Add `import * as Sentry from '@sentry/nextjs'` at top
   - In the existing `useEffect`, add `Sentry.captureException(error)` BEFORE the console.error call
   - Keep all existing UI markup unchanged

7. Update `apps/web/next.config.js`:
   - Add `const { withSentryConfig } = require('@sentry/nextjs');` at top
   - Change `module.exports = nextConfig;` to:
     ```javascript
     module.exports = withSentryConfig(nextConfig, {
       // Upload source maps for readable stack traces
       org: process.env.SENTRY_ORG,
       project: process.env.SENTRY_PROJECT,
       // Auth token for source map uploads
       authToken: process.env.SENTRY_AUTH_TOKEN,
       // Hide source maps from client bundles in production
       hideSourceMaps: true,
       // Silence source map upload logs unless there's an error
       silent: !process.env.CI,
     });
     ```

8. Add to `apps/web/.env.example` (append, do not remove existing vars):
   ```
   # Sentry Monitoring
   NEXT_PUBLIC_SENTRY_DSN=
   SENTRY_AUTH_TOKEN=
   SENTRY_ORG=
   SENTRY_PROJECT=
   ```
