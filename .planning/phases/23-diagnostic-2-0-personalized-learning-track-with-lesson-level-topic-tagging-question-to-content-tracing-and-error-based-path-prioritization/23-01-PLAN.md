export function parseLearningPath(lessons: unknown): string[] | SectionedLearningPath {
  if (Array.isArray(lessons)) return lessons; // old format: string[]
  if (typeof lessons === 'object' && lessons !== null && 'version' in lessons && (lessons as any).version === 2) {
    return lessons as SectionedLearningPath;
  }
  return []; // fallback
}

**1. Create `packages/ai/src/tagging.ts`** — LLM tagging logic:

```typescript
import { z } from 'zod';
import { openrouter, MODELS } from './openrouter';
import { supabase } from './retrieval';

export const lessonTagSchema = z.object({
  skillCategories: z.array(z.enum(['ANALYTICS', 'MARKETING', 'CONTENT', 'OPERATIONS', 'FINANCE'])).min(1).max(3),
  topics: z.array(z.string().min(2).max(50)).min(2).max(5),
  difficulty: z.enum(['EASY', 'MEDIUM', 'HARD']),
});

export type LessonTag = z.infer<typeof lessonTagSchema>;
```

Implement `tagLesson(lessonId: string, chunks: string[]): Promise<LessonTag>`:
- System prompt (exact text from RESEARCH.md Pattern 1 "LLM Tagging Prompt" — the full prompt about MPSTATS Academy with all 5 category definitions including FINANCE guidance for unit economics/margins/ROI)
- User prompt: concatenated chunk content
- Model: `MODELS.chat` (gemini-2.5-flash), fallback to `MODELS.fallback`
- `response_format: { type: 'json_schema' }` with strict schema matching lessonTagSchema
- Parse response, validate with Zod, return LessonTag
- JSON Schema for response_format:
```json
{
  "type": "object",
  "additionalProperties": false,
  "required": ["skillCategories", "topics", "difficulty"],

    "skillCategories": { "type": "array", "items": { "type": "string", "enum": ["ANALYTICS", "MARKETING", "CONTENT", "OPERATIONS", "FINANCE"] }, "minItems": 1, "maxItems": 3 },
    "topics": { "type": "array", "items": { "type": "string", "minLength": 2, "maxLength": 50 }, "minItems": 2, "maxItems": 5 },

  }
}
```

Implement `fetchLessonChunks(lessonId: string, limit = 3): Promise<string[]>`:
- Query `content_chunk` via supabase: `select('content').eq('lesson_id', lessonId).order('timecode_start', { ascending: true }).limit(limit)`
- Return array of content strings

Implement `clusterTopics(rawTopics: string[]): Promise<Record<string, string>>`:
- Takes all unique raw topics from stage 1 (de-duped)
- Single LLM call with system prompt: "Group these Russian educational topics into canonical forms. Merge synonyms (e.g. 'маржа', 'маржинальность', 'маржинальная прибыль' -> 'Маржинальность'). Return JSON: { 'raw_topic': 'canonical_topic', ... } for ALL input topics."
- User prompt: JSON array of all raw topics
- Parse response as `Record<string, string>`
- Target: 50-100 canonical topics from ~1000 raw topics

**2. Create `scripts/tag-lessons.ts`** — executable script:

```typescript
#!/usr/bin/env npx tsx
```

Steps:
1. Initialize Prisma client (import from `@mpstats/db`)
2. Fetch all lessons: `prisma.lesson.findMany({ select: { id: true, title: true, courseId: true } })`
3. **Stage 1**: For each lesson (sequential, 1-second delay between calls):
   - `fetchLessonChunks(lesson.id, 3)` — get first 3 chunks
   - If no chunks found, skip lesson (log warning)
   - `tagLesson(lesson.id, chunks)` — get LLM tags
   - Collect all raw topics
   - Update lesson in DB: `prisma.lesson.update({ where: { id }, data: { skillCategories: tag.skillCategories, topics: tag.topics, skillLevel: tag.difficulty } })`
   - Also update `skillCategory` (single) to `tag.skillCategories[0]` (primary category)
   - Log progress: `[42/405] 01_analytics_m01_start_001 -> [ANALYTICS, FINANCE] MEDIUM`
4. **Stage 2**: Topic clustering:
   - Collect all unique raw topics from stage 1
   - `clusterTopics(allRawTopics)` — get canonical mapping
   - For each lesson, replace raw topics with canonical versions: `prisma.lesson.update({ where: { id }, data: { topics: canonicalTopics } })`
5. **Summary stats**: Print count per category, count per difficulty, total canonical topics

Script must be idempotent (re-runnable — uses `update` not `create`).

Add to `package.json` root scripts:
```json
"tag-lessons": "npx tsx scripts/tag-lessons.ts"
```

**Environment:** Script uses `OPENROUTER_API_KEY` and `DATABASE_URL` from `.env` — same as existing scripts.
